🏆 Lightning McQueen Performance

Make your code faster than Lightning McQueen! Ka-chiga!

# Speed like Lightning McQueen
from functools import lru_cache

@lru_cache(maxsize=128)  # "Speed. I am speed."
def fibonacci_racing(n):
    if n < 2:
        return n
    return fibonacci_racing(n-1) + fibonacci_racing(n-2)

# Now it's lightning fast!
print(fibonacci_racing(100))
# Memory optimization - like WALL-E
def wall_e_efficiency():
    """Compact and efficient, just like our robot"""
    # Generator expression uses less memory
    return (x**2 for x in range(1000) if x % 2 == 0)

# Use it efficiently
squares = wall_e_efficiency()
first_five = [next(squares) for _ in range(5)]
